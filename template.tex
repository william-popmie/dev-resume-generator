% ---- Section heading with rule (no titlesec needed) ----
\renewcommand{\section}[1]{%
  \vspace{12pt}%
  {\large\bfseries\scshape #1}%
  \par\nobreak\vspace{-4pt}%
  \noindent\rule{\linewidth}{0.4pt}%
  \vspace{5pt}\par%
}

% ---- Resume commands ----
\newcommand{\resumeItem}[1]{%
  \item\small{#1}%
}

\newcommand{\resumeSubheading}[4]{%
  \item[]%
  \begin{tabular*}{\textwidth}[t]{l@{\extracolsep{\fill}}r}%
    \textbf{#1} & \small #2 \\%
    \textit{\small #3} & \textit{\small #4} \\%
  \end{tabular*}\vspace{-3pt}%
}

\newcommand{\resumeProjectHeading}[2]{%
  \item[]%
  \begin{tabular*}{\textwidth}{l@{\extracolsep{\fill}}r}%
    \small #1 & \small #2 \\%
  \end{tabular*}\vspace{-3pt}%
}

% Built-in \list instead of enumitem
\newcommand{\resumeSubHeadingListStart}{%
  \begin{list}{}{%
    \setlength{\leftmargin}{0pt}%
    \setlength{\itemsep}{8pt}%
    \setlength{\parsep}{0pt}%
    \setlength{\topsep}{2pt}%
  }%
}
\newcommand{\resumeSubHeadingListEnd}{\end{list}}

\newcommand{\resumeItemListStart}{%
  \begin{list}{\textbullet}{%
    \setlength{\leftmargin}{1.5em}%
    \setlength{\itemsep}{2pt}%
    \setlength{\parsep}{0pt}%
    \setlength{\topsep}{4pt}%
  }%
}
\newcommand{\resumeItemListEnd}{\end{list}\vspace{2pt}}
